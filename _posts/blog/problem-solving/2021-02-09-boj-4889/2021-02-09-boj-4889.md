#include <iostream>
#include <stack>

using namespace std;

int solution(string s) {
    int res = 0;

    stack<char> st;
    for (char each : s) {
        if (each == '{') st.push(each);
        else {
            if (st.empty() || st.top() != '{') {
                st.push('{');
                res++;
            }
            else st.pop();
        }
    }
    while (!st.empty()) {
        res++;
        st.pop();
        st.pop();
    }
    return res;
}
int main() {
    ios_base::sync_with_stdio(false);
	cin.tie(NULL);

    int caseNum = 1;
    while (1) {
        string s; cin >> s;
        if (s[0] == '-') break;

        cout << caseNum++ << ". " << solution(s) << "\n";
    }
}
